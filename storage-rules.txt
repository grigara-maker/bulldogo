// Firebase Storage pravidla - KOMPLETNÍ VERZE
// Tyto pravidla musí být nastavena v Firebase Console
// 
// DŮLEŽITÉ: Zkopírujte tento obsah do Firebase Console → Storage → Rules → Edit
// POZNÁMKA: Při testování v Rules simulátoru zvolte "Write" jako Simulation type

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // ===== OBRÁZKY SLUŽEB/INZERÁTŮ =====
    // Povolit přihlášeným uživatelům nahrávat obrázky služeb do jejich složek
    match /services/{userId}/{allPaths=**} {
      // Každý může číst obrázky
      allow read: if true;
      
      // Pouze vlastník (userId odpovídá UID) může nahrávat a upravovat
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // ===== PROFILOVÉ OBRÁZKY =====
    // Povolit přihlášeným uživatelům nahrávat profilové obrázky do jejich složek
    match /profile-pictures/{userId}/{allPaths=**} {
      // Každý může číst profilové obrázky
      allow read: if true;
      
      // Pouze vlastník (userId odpovídá UID) může nahrávat, upravovat a mazat
      allow write: if request.auth != null && request.auth.uid == userId;
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // ===== OBRÁZKY V CHATU =====
    // Povolit přihlášeným uživatelům nahrávat obrázky do chatu
    // Cesta: chat/{conversationId}/{timestamp}_{filename}
    match /chat/{conversationId}/{allPaths=**} {
      // Každý přihlášený uživatel může číst obrázky z konverzací
      allow read: if request.auth != null;
      
      // Každý přihlášený uživatel může nahrávat obrázky do konverzací
      allow write: if request.auth != null;
    }
    
    // ===== DALŠÍ SEKCE (přidej zde další, pokud je máš) =====
    // Příklad pro další složky:
    // match /documents/{userId}/{allPaths=**} {
    //   allow read: if true;
    //   allow write: if request.auth != null && request.auth.uid == userId;
    // }
    
    // ===== VÝCHOZÍ: ZAKÁZAT PŘÍSTUP =====
    // Všechno ostatní je zakázané
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
